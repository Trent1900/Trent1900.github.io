```js
const offset = { left: 0, right: 0, top: 0.1 * height, bottom: 0 };

  rect.top >= -offset.top &&
  rect.left >= -offset.left &&
  rect.bottom <= height + offset.bottom &&
  rect.right <= width + offset.right;
```
